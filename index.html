// This is a program to allow Owl to list however many terms
// in the Fibonacci series she wishes displayed on the screen

#include 
using namespace std;
int main()
{

  int fib_zero = 0;
  int fib_one = 1;
  int fib_two;
  int n;
  cout << "\n";
  cout << "Greetings, Owl!\n";
  cout << "Enter the number of Fibonacci numbers you'd like displayed.\n";
   cin >> n;
  cout << "\n"
       << "Your " 
       << n 
       << " " 
       << "Fibonacci numbers are displayed below, in order from left to right:\n"
       << fib_zero  // Outputting 0 
       << " "      // Spaces out each term 
       << fib_one // Outputting 1
       << " "; 
  for (int i = 2; i < n; i++) // Loop starts from 2 because 0 and 1 have already been outputted
                              // ++i is different from i++                             
    {                       
      fib_two = fib_zero + fib_one;
      fib_zero = fib_one;
      fib_one = fib_two; 
      cout << fib_two 
           << " ";
    }
  cout << "\n"
       << "Have an awesome day, Owl!\n" 
       << endl;
  return 0;
}
