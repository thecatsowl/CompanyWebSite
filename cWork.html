/************************************************************************************************************************
    Instructions: 
    
    Demonstrate the power of exponential growth in a C program: Picture a chessboard. Begin with one grain
    of wheat on the first square, two on the second, 4 on the third, and so on, until you've doubled 
    each preceding grain of wheat, stopping at the 64th square on the chessboard. 
			  
    Write a program that calculates how many grains go on each square and keeps a running total. 
			  
    A very rough estimate of the annual world wheat production in wheat grains is 2E16 in e-notation 
    (2 x 10^16 in scientific notation or 20,000,000,000,000,000 wheat grains; that's 20 quadrillion.)
			  
    Include a comparison of the running total as a fraction of the world total, i.e., the world wheat 
    production in grains.
			  
    By the time you've reached the 55th square, you've exceeded the annual world wheat production!

    This program shows the power of exponential growth, which is the method by which the COVID-19 virus has 
    been able to propagate so efficiently.

    Below is the code for this program.

************************************************************************************************************************/

#include <stdio.h>
#define SQUARES 64	// number of squares on a chessboard

int main(void)
{

	const double CROP = 2E16;	// world wheat production in grains
	double current, total;
	int i=1;	// counter to keep track of iterations in while loop

	printf("\nsquare	   grains	total	");
	printf("    fraction of \n");
	printf("	   added	grains	");
	printf("    world total\n");
	printf("**************************************************\n");
	total = current = 1.0;	// start with one grain
	printf("%4d %13.2e %12.2e %14.2e \n", i, current, total, total/CROP);
	while(i<SQUARES)
	{

		i += 1;
		current *= 2.0;	// double grains on next square
		total += current;	// update total
		printf("%4d %13.2e %12.2e %14.2e\n", i, current, total, total/CROP);
	}
	printf("**************************************************\n");

	printf("\n               Stay safe and goodbye.\n\n");

	return 0;
}
